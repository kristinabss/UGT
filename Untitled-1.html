Пример генерации отчета (заглушка) product = next((p for p in products_db if p.id == product_id), None) ugt = next((r for r in reversed(ugt_scores_db) if r.product_id == product_id), None) if not (product and ugt): raise HTTPException(status_code=404, detail="No data") return UGTReport( product_id=product.id, product_name=product.name, technology=ugt.technology, ugt_score=ugt.score, details="Демо-отчет о технологической готовности" ) #
